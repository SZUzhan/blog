    # [Js(v8)]

    # 输入

    # 读取一行输入

    read_line()，函数别名： readline()、readLine()

    将读取至多1024个字符，当还未达到1024个时如果遇到回车或结束符，提前结束。

    读取多行最简单的办法是while((line = read_line()) != '')。

    # 读取多行输入

    //比如以下输入
//10
//2 6 5 6 5 6 5 2 23 99

let n = Number(readline())
let arr = readline()
while((line = readline()).length>0) {
  arr += line
}
arr = arr.split(' ').map(Number)//读取的都是字符串，转为数组

    # JS大数运算

    JS 中的Number类型只能安全地表示**-9007199254740991 (-(2^53-1))** 和**9007199254740991(2^53-1)**之间的整数，任何超出此范围的整数值都可能失去精度。

    BigInt的存在是为了弥补Number表示的缺陷，常规情况下，使用Number就能满足我们所有的需求，但如果是处理超高精度的计算，或者加密、时间戳等任务就需要使用BigInt类型。

    //尾部加n表示为BigInit类型
const bigInt = 999999999999999999999999999999999999999999999n;
//也可以直接用BigInt
let bignum = BigInt(input);

console.log(typeof bignum);//'bigint'

function transposeMatrix(input) {
    let n = input.length;
    let m = input[0].length;
    let result = [];
    for (let i = 0; i < m; i++) {

    # [Nodejs]

    需要引入node-stdio包。

    const stdio = require('node-stdio');

    # 读取一行输入

    stdio.read_line()，函数别名： stdio.readline()、stdio.readLine()

    将读取至多1024个字符，当还未达到1024个时如果遇到回车或结束符，提前结束。

    读取多行最简单的办法是while((line = stdio.read_line()) != '')。

//10
//2 6 5 6 5 6 5 2 23 99

const stdio = require('node-stdio');
